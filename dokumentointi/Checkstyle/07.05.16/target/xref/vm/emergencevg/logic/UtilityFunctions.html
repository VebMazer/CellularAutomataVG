26      /**
27       * Etsii viimeisenä particleTypes listaan laitetun olion avaimen, joka on
28       * siis aina suurin avainten joukossa oleva luku.
29       *
30       * @return Palauttaa suurimman partikkelityyppimapin avaimen.
31       */
32      public int findLatestKey() {
33          int lastKey = 1;
34          for (int key : space.particleTypes.keySet()) {
35              if (key > lastKey) {
36                  lastKey = key;
37              }
38          }
39          return lastKey;
40      }

42      /**
43       * Tarkistaa onko merkki numero.
44       *
45       * @param c merkki
46       * @return totuusarvo
47       */
48      public boolean checkIfNumber(char c) {
49          return (c >= '0' && c <= '9');
50      }

52      /**
53       * Parseroi merkkijonosta indeksin jälkeen seuraavan numeron ja liittää sen
54       * listaan ja palauttaa numeron jälkeen seuraavan indeksin.
55       *
56       * @param index Indeksi josta merkkijonon läpikäynti alkaa.
57       * @param str Merkkijono jota käydään läpi-
58       * @param list Lista johon löydetty numero listään.
59       * @return Numeroa seuraavan indeksin, tai merkkijonon pituuden.
60       */
61      public int parseNextNumber(int index, String str, ArrayList<Integer> list) {
62          while (index < str.length()) {
63              if (space.uFunctions.checkIfNumber(str.charAt(index))) {
64                  break;
65              }
66              index++;
67          }
68          int beginIndex = index;
69          while (index < str.length()) {
70              index++;
71              if (!space.uFunctions.checkIfNumber(str.charAt(index))) {
72                  break;
73              }
74          }
75          list.add(Integer.parseInt(str.substring(beginIndex, index)));
76          return index;
77      }

79      /**
80       * Parseroi merkkijonosta numeron.
81       *
82       * @param str merkkijono
83       * @param index numeron alkuun viittaava indeksi.
84       * @return int arvo
85       */
86      public int parseNumber(String str, int index) {
87          int startIndex = index;
88          while (index < str.length() && str.charAt(index) != ',' && str.charAt(index) != ')' && str.charAt(index) != ' ') {
89              index++;
90          }
91          return Integer.parseInt(str.substring(startIndex, index));
92      }

100     public int initializeCommandArrayList() {
101         int iteration = coReRunner.iterations;
102         if (!coReRunner.commandsToBeAdded.containsKey(iteration)) {
103             coReRunner.commandsToBeAdded.put(iteration, new ArrayList<String>());
104         }
105         return iteration;

126     /**
127      * Lisää lisätyt komennot komento mappiin.
128      */
129     public void uniteCommandMaps() {
130         for (int key : coReRunner.commandsToBeAdded.keySet()) {
131             if (coReRunner.commands.containsKey(key)) {
132                 ArrayList<String> commandList = coReRunner.commands.get(key);
133                 for (String command : coReRunner.commandsToBeAdded.get(key)) {
134                     commandList.add(command);
135                 }
136             } else {
137                 coReRunner.commands.put(key, coReRunner.commandsToBeAdded.get(key));
138             }
139         }
140         coReRunner.commandsToBeAdded = new HashMap<Integer, ArrayList<String>>();
141     }

153     /**
154      * Laskee yleisimmän int arvon, joka ei ole nolla ja palauttaa sen avaimen.
155      * Palauttaa 0 jos yhtäkään partikkelia ei löydy. Metodi suosii yhtä monen
156      * arvon tilanteissa pienintä arvoa(Listan ensimmäinen partikkeli.)
157      *
158      * @param keys Taulukko, joka sisältää int muuttujia.
159      * @return Taulukon yleisin int arvo, joka ei ole nolla, tai nolla jos muita
160      * arvoja ei ole.
161      */
162     public int mostCommonKey(int[] keys) {
163         Arrays.sort(keys);
164         int mostRep = 0;
165         int reps = 0;
166         int index = 0;
167         boolean grew = false;
168         for (int i = 0; i < keys.length; i++) {
169             if (keys[i] != 0) {
170                 if (reps < 1) {
171                     reps = 1;
172                 }
173                 if (i + 1 < keys.length && keys[i] == keys[i + 1]) {
174                     reps++;
175                     grew = true;
176                 } else {
177                     grew = false;
178                 }
179                 if (reps > mostRep) {
180                     mostRep = reps;
181                     index = i;
182                 }
183                 if (!grew) {
184                     reps = 0;
185                 }
186             }
187         }
188         return keys[index];
189     }
