3   import java.awt.Color;
4   import java.awt.Graphics;
5   import javax.swing.JPanel;
6   import vm.emergencevg.logic.GenerativeSpace;
7   
8   public class DrawBoard extends JPanel implements Updateable {
9   
10      GenerativeSpace space;
11      int sideLength;
12  
13      public DrawBoard(GenerativeSpace space, int sideLength) {
14          this.space = space;
15          this.sideLength = sideLength;
16  
17      }
18  
19      protected void paintComponent(Graphics g) {
20          super.paintComponent(g);
21  
22          g.setColor(Color.RED);
23          for (int j = 0; j < space.ylength; j++) {
24              for (int i = 0; i < space.xlength; i++) {
25                  if (space.field[i][j] != 0) {
26                      g.fill3DRect(i * sideLength, j * sideLength, sideLength, sideLength, true);
27                  }
28              }
29  
30          }
31      }
32  
33      @Override
34      public synchronized void update() {
35          repaint();
36      }
37  }
